### Compare the Authoritarian-Libertarian axis

country_event = {
  id = tyranny_majority.100

  is_triggered_only = yes
  hide_window = yes

  immediate = {
    set_variable = {
      which = pops_fanatic_collectivist
      value = 0
    }
    set_variable = {
      which = pops_collectivist
      value = 0
    }
    set_variable = {
      which = total_collectivists
      value = 0
    }
    set_variable = {
      which = pops_individualist
      value = 0
    }
    set_variable = {
      which = pops_fanatic_individualist
      value = 0
    }
    set_variable = {
      which = total_individualists
      value = 0
    }
    set_variable = {
      which = pops_neutral_col_ind
      value = 0
    }
    set_variable = {
      which = pops_not_neutral_col_ind
      value = 0
    }
    save_event_target_as = country_col_ind

    # Count Fanatic Authoritarians
    every_owned_pop = {
      limit = { has_ethic = "ethic_fanatic_collectivist" }
      event_target:country_col_ind = {
        change_variable = {
          which = pops_fanatic_collectivist
          value = 1
        }
        change_variable = {
          which = total_collectivists
          value = 1
        }
        change_variable = {
          which = pops_not_neutral_col_ind
          value = 1
        }
      }
    }
    # Count Authoritarians
    every_owned_pop = {
      limit = { has_ethic = "ethic_collectivist" }
      event_target:country_col_ind = {
        change_variable = {
          which = pops_collectivist
          value = 1
        }
        change_variable = {
          which = total_collectivists
          value = 1
        }
        change_variable = {
          which = pops_not_neutral_col_ind
          value = 1
        }
      }
    }
    # Count Libertarians
    every_owned_pop = {
      limit = { has_ethic = "ethic_individualist" }
      event_target:country_col_ind = {
        change_variable = {
          which = pops_individualist
          value = 1
        }
        change_variable = {
          which = total_individualists
          value = 1
        }
        change_variable = {
          which = pops_not_neutral_col_ind
          value = 1
        }
      }
    }
    # Count Fanatic Libertarians
    every_owned_pop = {
      limit = { has_ethic = "ethic_fanatic_individualist" }
      event_target:country_col_ind = {
        change_variable = {
          which = pops_fanatic_individualist
          value = 1
        }
        change_variable = {
          which = total_individualists
          value = 1
        }
        change_variable = {
          which = pops_not_neutral_col_ind
          value = 1
        }
      }
    }
    # Count neutrals
    every_owned_pop = {
      limit = {
        NOR {
          has_ethic = "ethic_fanatic_collectivist"
          has_ethic = "ethic_collectivist"
          has_ethic = "ethic_individualist"
          has_ethic = "ethic_fanatic_individualist"
        }
      }
      event_target:country_col_ind = {
        change_variable = {
          which = pops_neutral_col_ind
          value = 1
        }
      }
    }

    # If not neutrals > neutrals
    if {
      limit = {
        check_variable = {
          which = pops_not_neutral_col_ind
          value > pops_neutral_col_ind
        }
      }
      # If authoritarians > libertarians
      if = {
        limit = {
          check_variable = {
            which = total_collectivists
            value > total_individualists
          }
        }
        # If fanatics > moderates
        if = {
          limit = {
            check_variable = {
              which = pops_fanatic_collectivist
              value > pops_collectivist
            }
          }
          country_event = { id = tyranny_majority.10 }
          else = {
            country_event = { id = tyranny_majority.11 }
          }
        }
        # If libertarians > authoritarians
        else = {
          # If fanatics > moderates
          if = {
            limit = {
              check_variable = {
                which = pops_fanatic_individualist
                value > pops_individualist
              }
            }
            country_event = { id = tyranny_majority.13 }
            else = {
              country_event = { id = tyranny_majority.12 }
            }
          }
        }
      }
      else {
        country_event = { id = tyranny_majority.14 }
      }
